```python
# before
def _deprecated(f):
    def wraps(self, *args, **kwargs):
        # ... builds message ...
        logger.warning(message, *logger_args)  # fires every request

# after
def _deprecated(f):
    _warned = False  # ← closure flag per endpoint
    def wraps(self, *args, **kwargs):
        nonlocal _warned
        if not _warned:
            _warned = True
            # ... builds message ...
            logger.warning(message, *logger_args)  # fires once per process
        return f(self, *args, **kwargs)
```
